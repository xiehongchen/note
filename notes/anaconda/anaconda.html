# 1、查看python 环境

conda info --env
1

# 2、创建python环境

conda create --name python35 python=3.5
1
  • python35：环境名，可以更改
  • python=3.5：python版本

# 3、激活使用python

conda activate python35
1

# 4、给当前环境安装包

  • 不要使用conda进行下载，使用pip进行下载
pip install numpy
1

# 5、删除python环境

 conda remove -n python35 --all
1
